 55def subphase_detector_func(phase_id: str, phase_iteration_range: str, phase_iterations: list) -> str:
 56    """
 57    Detect subphases within a single phase of conversation iterations.
 58    
 59    Args:
 60        phase_id: Phase identifier
 61        phase_iteration_range: Range of iterations in this phase (e.g., '1-15')
 62        phase_iterations: Iterations within the specific phase
 63    """
 64    # Validate arguments using Pydantic model internally
 65    validated_args = SubphaseDetectorArgs(
 66        phase_id=phase_id,
 67        phase_iteration_range=phase_iteration_range,
 68        phase_iterations=phase_iterations
 69    )
 70    
 71    phase_id = validated_args.phase_id
 72    phase_iteration_range = validated_args.phase_iteration_range
 73    phase_iterations = validated_args.phase_iterations
 74    
 75    # Parse iteration range
 76    start_iter, end_iter = map(int, phase_iteration_range.split('-'))
 77    total_iterations = len(phase_iterations)
 78    
 79    # Analyze iterations to detect subphases
 80    subphases = []
 81    subphase_number = 1
 82    
 83    # Group iterations into subphases based on activity patterns
 84    # Look for shifts in activities: design → implementation → debugging → testing
 85    activity_patterns = {
 86        'design': ['design', 'architecture', 'plan', 'concept', 'structure'],
 87        'implementation': ['implement', 'create', 'build', 'develop', 'code'],
 88        'debugging': ['debug', 'fix', 'error', 'bug', 'issue', 'problem'],
 89        'testing': ['test', 'verify', 'validate', 'check', 'confirm'],
 90        'documentation': ['document', 'doc', 'readme', 'comment', 'explain'],
 91        'refactoring': ['refactor', 'optimize', 'improve', 'enhance', 'cleanup']
 92    }
 93    
 94    # Classify each iteration by primary activity
 95    iteration_activities = []
 96    for i, iteration in enumerate(phase_iterations):
 97        iter_text = iteration.lower()
 98        activity_scores = {}
 99        
100        for activity, keywords in activity_patterns.items():
101            score = sum(1 for keyword in keywords if keyword in iter_text)
102            activity_scores[activity] = score
103        
104        # Get primary activity (highest score)
105        primary_activity = max(activity_scores, key=activity_scores.get) if max(activity_scores.values()) > 0 else 'general'
106        iteration_activities.append(primary_activity)
107    
108    # Group consecutive iterations with similar activities into subphases
109    current_activity = iteration_activities[0]
110    subphase_start = 0
111    
112    for i in range(1, len(iteration_activities)):
113        # If activity changes or we reach a reasonable subphase size, create a subphase
114        if (iteration_activities[i] != current_activity) or (i - subphase_start >= 6):  # Max 6 iterations per subphase
115            # Create subphase for previous group
116            subphase_end = i - 1
117            actual_start = start_iter + subphase_start
118            actual_end = start_iter + subphase_end
119            
120            subphase_iterations = phase_iterations[subphase_start:i]
121            
122            # Generate subphase description
123            activity_name = current_activity.title()
124            if current_activity == 'general':
125                activity_name = 'General Development'
126            
127            subphases.append({
128                "subphase_number": subphase_number,
129                "iteration_range": f"{actual_start}-{actual_end}",
130                "activity_type": activity_name,
131                "description": f"Subphase {phase_id.split('_')[-1]}{chr(96 + subphase_number)} (iter {actual_start}-{actual_end}): {activity_name}",
132                "iteration_count": len(subphase_iterations)
133            })
134            
135            # Start new subphase
136            current_activity = iteration_activities[i]
137            subphase_start = i
138            subphase_number += 1
139    
140    # Handle the last subphase
141    if subphase_start < len(iteration_activities):
142        actual_start = start_iter + subphase_start
143        actual_end = start_iter + len(iteration_activities) - 1
144        subphase_iterations = phase_iterations[subphase_start:]
145        
146        activity_name = current_activity.title()
147        if current_activity == 'general':
148            activity_name = 'General Development'
149        
150        subphases.append({
151            "subphase_number": subphase_number,
152            "iteration_range": f"{actual_start}-{actual_end}",
153            "activity_type": activity_name,
154            "description": f"Subphase {phase_id.split('_')[-1]}{chr(96 + subphase_number)} (iter {actual_start}-{actual_end}): {activity_name}",
155            "iteration_count": len(subphase_iterations)
156        })
157    
158    # Format subphase detection results
159    result = f"""# Subphase Detection Results for {phase_id}
160
161**Phase Iteration Range:** {phase_iteration_range}
162**Total Iterations in Phase:** {total_iterations}
163**Subphases Detected:** {len(subphases)}
164
165## Subphase Breakdown:
166"""
167    
168    for subphase in subphases:
169        result += f"""
170### {subphase['description']}
171- **Activity Type:** {subphase['activity_type']}
172- **Iterations:** {subphase['iteration_range']} ({subphase['iteration_count']} iterations)
173"""
174    
175    result += f"""
176
177## Analysis Summary
178This phase was broken down into {len(subphases)} distinct subphases based on activity patterns and natural workflow boundaries. Each subphase represents a coherent unit of work within the larger phase.
179"""
180    
181    return result
