 46def phase_detector_func(conversation_id: str, iteration_summaries: list) -> str:
 47    """
 48    Detect logical phases in a conversation from iteration summaries.
 49    
 50    Args:
 51        conversation_id: Unique conversation identifier
 52        iteration_summaries: All iteration summaries from conversation
 53    """
 54    # Validate arguments using Pydantic model internally
 55    validated_args = PhaseDetectorArgs(
 56        conversation_id=conversation_id,
 57        iteration_summaries=iteration_summaries
 58    )
 59    
 60    conversation_id = validated_args.conversation_id
 61    iteration_summaries = validated_args.iteration_summaries
 62    
 63    # Analyze iteration summaries to detect phases
 64    total_iterations = len(iteration_summaries)
 65    
 66    # Simple phase detection logic - in production this would be more sophisticated
 67    # For now, group iterations by similar themes/topics
 68    phases = []
 69    current_phase_start = 1
 70    phase_number = 1
 71    
 72    # Group iterations into phases of roughly 10-15 iterations each
 73    # or when topic shifts are detected
 74    phase_size = min(15, max(5, total_iterations // 4))  # Adaptive phase sizing
 75    
 76    for i in range(0, total_iterations, phase_size):
 77        phase_end = min(i + phase_size, total_iterations)
 78        phase_iterations = iteration_summaries[i:phase_end]
 79        
 80        # Extract common themes from this phase
 81        phase_summary = f"Phase {phase_number} (iter {i+1}-{phase_end})"
 82        
 83        # Simple theme detection - look for common keywords
 84        phase_keywords = set()
 85        for summary in phase_iterations:
 86            # Extract potential keywords (this is simplified)
 87            words = summary.lower().split()
 88            potential_keywords = [w for w in words if len(w) > 4 and w.isalpha()]
 89            phase_keywords.update(potential_keywords[:3])  # Top 3 keywords per iteration
 90        
 91        top_keywords = list(phase_keywords)[:5]  # Top 5 keywords for this phase
 92        phase_description = f": {', '.join(top_keywords)} discussion" if top_keywords else ": General development work"
 93        
 94        phases.append({
 95            "phase_number": phase_number,
 96            "iteration_range": f"{i+1}-{phase_end}",
 97            "description": phase_summary + phase_description,
 98            "iteration_count": len(phase_iterations)
 99        })
100        
101        phase_number += 1
102    
103    # Format phase detection results
104    result = f"""# Phase Detection Results for {conversation_id}
105
106**Total Iterations Analyzed:** {total_iterations}
107**Phases Detected:** {len(phases)}
108
109## Phase Breakdown:
110"""
111    
112    for phase in phases:
113        result += f"""
114### {phase['description']}
115- **Iterations:** {phase['iteration_range']} ({phase['iteration_count']} iterations)
116"""
117    
118    result += f"""
119
120## Summary
121This conversation had {len(phases)} distinct phases across {total_iterations} iterations. Each phase represents a logical grouping of related work or discussion topics.
122"""
123    
124    return result
