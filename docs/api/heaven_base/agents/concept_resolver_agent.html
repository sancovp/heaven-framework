 38        """Complex semantic validation for concept quality"""
 39        if not v:
 40            raise ValueError('preliminary_concepts cannot be empty')
 41

 43            if not phrase.strip():
 44                raise ValueError('preliminary_concepts cannot contain empty strings')

 49

 54            # Validate sanctuary references need context
 55            if 'sanctuary' in phrase.lower():
 56                valid_contexts = ['project', 'system', 'feature', 'implementation', 'design']
 57                if not any(context in phrase.lower() for context in valid_contexts):

 63def concept_resolver_func(phase_summaries: list, preliminary_concepts: list) -> str:
 64    """
 65    Resolve canonical concepts from complete conversation context.
 66    
 67    Args:
 68        phase_summaries: Rich phase summaries with complete context from Phase Aggregator
 69        preliminary_concepts: Preliminary concept phrases to validate and resolve
 70    """
 71    # Validate arguments using Pydantic model internally
 72    validated_args = ConceptResolverArgs(
 73        phase_summaries=phase_summaries,
 74        preliminary_concepts=preliminary_concepts
 75    )
 76    
 77    conversation_id = validated_args.conversation_id
 78    phase_summaries = validated_args.phase_summaries
 79    preliminary_concepts = validated_args.preliminary_concepts
 80    metadata = validated_args.metadata
 81    
 82    # Analyze complete conversation context for concept resolution
 83    total_phases = len(phase_summaries)
 84    
 85    # Build comprehensive context understanding
 86    conversation_themes = set()
 87    technical_entities = set()
 88    process_entities = set()
 89    
 90    # Extract entities and themes from all phases
 91    for phase_summary in phase_summaries:
 92        phase_text = phase_summary.lower()
 93        
 94        # Extract technical entities (files, systems, tools)
 95        tech_patterns = [
 96            r'(\w+\.py)',           # Python files
 97            r'(\w+_\w+_system)',    # System names
 98            r'(\w+tool)',           # Tools
 99            r'(\w+agent)',          # Agents
100            r'(heaven\w*)',         # HEAVEN-related
101        ]
102        
103        for pattern in tech_patterns:
104            matches = re.findall(pattern, phase_text)
105            technical_entities.update(matches)
106        
107        # Extract process entities (activities, phases)
108        if 'debug' in phase_text:
109            process_entities.add('debugging_process')
110        if 'implement' in phase_text:
111            process_entities.add('implementation_process')
112        if 'design' in phase_text:
113            process_entities.add('design_process')
114        if 'test' in phase_text:
115            process_entities.add('testing_process')
116    
117    # Resolve preliminary concepts into canonical forms
118    canonical_concepts = {}
119    concept_relationships = []
120    
121    for concept in preliminary_concepts:
122        concept_lower = concept.lower()
123        
124        # Resolve specific patterns
125        if 'sanctuary' in concept_lower:
126            if 'project' in concept_lower or 'system' in concept_lower:
127                canonical_form = "SANCTUARY_project"
128                canonical_concepts[concept] = {
129                    "canonical_form": canonical_form,
130                    "type": "project_entity",
131                    "description": "The SANCTUARY project discussed across multiple phases",
132                    "phases_mentioned": [i+1 for i, summary in enumerate(phase_summaries) if 'sanctuary' in summary.lower()]
133                }
134        
135        elif 'auto_summarize' in concept_lower or 'summarize' in concept_lower:
136            if 'error' in concept_lower or 'bug' in concept_lower:
137                canonical_form = "auto_summarize_system_debugging"
138                canonical_concepts[concept] = {
139                    "canonical_form": canonical_form,
140                    "type": "debugging_activity",
141                    "description": "Debugging activities related to the auto_summarize system",
142                    "phases_mentioned": [i+1 for i, summary in enumerate(phase_summaries) if 'summarize' in summary.lower() and ('error' in summary.lower() or 'debug' in summary.lower())]
143                }
144            else:
145                canonical_form = "auto_summarize_system"
146                canonical_concepts[concept] = {
147                    "canonical_form": canonical_form,
148                    "type": "technical_system",
149                    "description": "The auto_summarize system and its components",
150                    "phases_mentioned": [i+1 for i, summary in enumerate(phase_summaries) if 'summarize' in summary.lower()]
151                }
152        
153        elif 'concept' in concept_lower and 'tag' in concept_lower:
154            canonical_form = "concept_tagging_system"
155            canonical_concepts[concept] = {
156                "canonical_form": canonical_form,
157                "type": "feature_implementation",
158                "description": "Implementation and improvement of concept tagging functionality",
159                "phases_mentioned": [i+1 for i, summary in enumerate(phase_summaries) if 'concept' in summary.lower() or 'tag' in summary.lower()]
160            }
161        
162        elif 'rate' in concept_lower and 'limit' in concept_lower:
163            canonical_form = "rate_limiting_configuration"
164            canonical_concepts[concept] = {
165                "canonical_form": canonical_form,
166                "type": "system_configuration",
167                "description": "Rate limiting issues and configuration adjustments",
168                "phases_mentioned": [i+1 for i, summary in enumerate(phase_summaries) if 'rate' in summary.lower() or 'limit' in summary.lower()]
169            }
170        
171        elif 'aimessage' in concept_lower or 'ai_message' in concept_lower:
172            if 'import' in concept_lower or 'error' in concept_lower:
173                canonical_form = "aimessage_import_bug"
174                canonical_concepts[concept] = {
175                    "canonical_form": canonical_form,
176                    "type": "technical_bug",
177                    "description": "AIMessage import-related bugs and fixes",
178                    "phases_mentioned": [i+1 for i, summary in enumerate(phase_summaries) if 'aimessage' in summary.lower() or 'import' in summary.lower()]
179                }
180        
181        else:
182            # Generic resolution for other concepts
183            canonical_form = concept.lower().replace(' ', '_').replace('-', '_')
184            canonical_concepts[concept] = {
185                "canonical_form": canonical_form,
186                "type": "general_concept",
187                "description": f"General concept: {concept}",
188                "phases_mentioned": [i+1 for i, summary in enumerate(phase_summaries) if any(word in summary.lower() for word in concept.lower().split())]
189            }
190    
191    # Create relationships between concepts
192    for concept1, data1 in canonical_concepts.items():
193        for concept2, data2 in canonical_concepts.items():
194            if concept1 != concept2:
195                # Check for phase overlap
196                phase_overlap = set(data1["phases_mentioned"]) & set(data2["phases_mentioned"])
197                if phase_overlap:
198                    relationship = {
199                        "from_concept": data1["canonical_form"],
200                        "to_concept": data2["canonical_form"],
201                        "relationship_type": "co_occurred",
202                        "shared_phases": list(phase_overlap),
203                        "strength": len(phase_overlap) / max(len(data1["phases_mentioned"]), len(data2["phases_mentioned"]))
204                    }
205                    concept_relationships.append(relationship)
206    
207    # Format concept resolution results
208    result = f"""# Concept Resolution Results for {conversation_id}
209
210**Total Phases Analyzed:** {total_phases}
211**Preliminary Concepts Processed:** {len(preliminary_concepts)}
212**Canonical Concepts Resolved:** {len(canonical_concepts)}
213**Concept Relationships Identified:** {len(concept_relationships)}
214
215## Canonical Concepts:
216"""
217    
218    for original_concept, resolution_data in canonical_concepts.items():
219        result += f"""
220### {resolution_data['canonical_form']} 
221- **Original:** {original_concept}
222- **Type:** {resolution_data['type']}
223- **Description:** {resolution_data['description']}
224- **Mentioned in Phases:** {', '.join(map(str, resolution_data['phases_mentioned']))}
225"""
226    
227    result += f"""
228
229## Concept Relationships:
230"""
231    
232    # Show top relationships by strength
233    top_relationships = sorted(concept_relationships, key=lambda x: x["strength"], reverse=True)[:10]
234    for rel in top_relationships:
235        result += f"""
236- **{rel['from_concept']}** ↔ **{rel['to_concept']}** 
237  - Relationship: {rel['relationship_type']}
238  - Shared Phases: {', '.join(map(str, rel['shared_phases']))}
239  - Strength: {rel['strength']:.2f}
240"""
241    
242    result += f"""
243
244## Resolution Summary
245Successfully resolved {len(canonical_concepts)} concepts from the complete conversation context. The canonical concepts provide a clean, queryable knowledge representation with {len(concept_relationships)} identified relationships for graph storage.
246
247## Technical Details
248- Conversation ID: {conversation_id}
249- Processing Date: Generated with complete conversation context
250- Quality: High-fidelity concept resolution with semantic validation
251"""
252    
253    return result
