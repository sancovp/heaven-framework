  67async def completion_runner(state: HeavenState, *, node_config: Dict[str, Any] = None, prompt: str = None, agent=None, target_container: str = None, source_container: str = None, history_id: str = None, prompt_template_vars: Dict[str, Any] = None) -> Dict[str, Any]:
  68    """
  69    Execute completion-style agent and store raw result.
  70    
  71    Can be called with normal parameters or legacy node_config:
  72    
  73    Normal way:
  74        completion_runner(state, agent=my_agent, prompt="Hello")
  75    
  76    Legacy way:
  77        completion_runner(state, node_config={"agent": my_agent, "prompt": "Hello"})
  78    """
  79    # Build node_config from parameters if not provided
  80    if node_config is None:
  81        if prompt is None:
  82            raise ValueError("Must provide either node_config or prompt parameter")
  83        node_config = {
  84            "prompt": prompt
  85        }
  86        if agent is not None:
  87            node_config["agent"] = agent
  88        if target_container is not None:
  89            node_config["target_container"] = target_container
  90        if source_container is not None:
  91            node_config["source_container"] = source_container
  92        if history_id is not None:
  93            node_config["history_id"] = history_id
  94        if prompt_template_vars:
  95            node_config["prompt_template_vars"] = prompt_template_vars
  96    
  97    # Get agent
  98    agent = node_config["agent"]
  99    if isinstance(agent, str):
 100        agent = state["agents"][agent]
 101    
 102    # Build prompt
 103    prompt = node_config["prompt"]
 104    if "prompt_template_vars" in node_config:
 105        # Simple template substitution
 106        for key, value in node_config["prompt_template_vars"].items():
 107            prompt = prompt.replace(f"{{{key}}}", str(value))
 108    
 109    # Execute with all parameters
 110    exec_params = {"prompt": prompt}
 111    if agent:
 112        exec_params["agent"] = agent
 113    if "target_container" in node_config:
 114        exec_params["target_container"] = node_config["target_container"]
 115    if "source_container" in node_config:
 116        exec_params["source_container"] = node_config["source_container"]
 117    if "history_id" in node_config:
 118        exec_params["history_id"] = node_config["history_id"]
 119    
 120    result = await exec_completion_style(**exec_params)
 121    
 122    # Store raw result
 123    execution_result = {
 124        "node_type": HeavenNodeType.COMPLETION,
 125        "agent_name": agent.name,
 126        "prompt": prompt,
 127        "raw_result": result
 128    }
 129    
 130    return {
 131        "results": state["results"] + [execution_result]
 132    }

1204        variables = variables or {}
1205        
1206        # Process nodes with variable substitution
1207        def substitute_variables(obj):
1208            """Recursively substitute {variable} placeholders with actual values"""
1209            if isinstance(obj, str):
1210                # Simple string substitution for {variable} patterns
1211                for var_name, var_value in variables.items():
1212                    placeholder = f"{{{var_name}}}"
1213                    if placeholder in obj:
1214                        # If the entire string is just the placeholder, return the object directly
1215                        if obj == placeholder:
1216                            return var_value
1217                        # Otherwise do string replacement
1218                        obj = obj.replace(placeholder, str(var_value))
1219                return obj
1220            elif isinstance(obj, dict):
1221                return {k: substitute_variables(v) for k, v in obj.items()}
1222            elif isinstance(obj, list):
1223                return [substitute_variables(item) for item in obj]
1224            else:
1225                return obj
1226        
1227        # Add nodes with variable substitution
1228        for node in spec.nodes:
1229            node_id = node["id"]
1230            node_type = node["type"]
1231            node_config = substitute_variables(node["config"])
1232            
1233            if node_type in self.node_runners:
1234                # Create node function with config bound using partial
1235                from functools import partial
1236                node_func = partial(self.node_runners[node_type], node_config=node_config)
1237                
1238                graph.add_node(node_id, node_func)
1239        
1240        # Add edges
1241        for edge in spec.edges:
1242            from_node = edge["from"]
1243            to_node = edge["to"]
1244            
1245            # Handle special node names
1246            if from_node == "START":
1247                from_node = START
1248            if to_node == "END":
1249                to_node = END
1250            
1251            graph.add_edge(from_node, to_node)
1252        
1253        # Compile first, then store variables and initial context
1254        compiled_graph = graph.compile()
1255        compiled_graph._heaven_variables = variables
1256        compiled_graph._heaven_initial_context = spec.initial_context
1257        
1258        return compiled_graph
1259    
1260    async def run_graph(self, compiled_graph, thread_id: str = "heaven_session") -> List[Dict[str, Any]]:
1261        """Run a compiled graph and return results"""
1262        
1263        # Get stored variables and initial context
1264        variables = getattr(compiled_graph, '_heaven_variables', {})
1265        initial_context = getattr(compiled_graph, '_heaven_initial_context', {})
1266        
1267        initial_state = {
1268            "results": [],
1269            "context": initial_context,
1270            "agents": {}  # No longer using agents in state, use variables directly
1271        }
1272        
1273        memory = MemorySaver()
1274        config = {"configurable": {"thread_id": thread_id}}
1275        
1276        result = await compiled_graph.ainvoke(initial_state, config)
1277        
1278        # Store final state for subgraph access
1279        self.final_state = result
1280        
1281        return result.get("results", [])
