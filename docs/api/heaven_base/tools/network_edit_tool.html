  86def clean_networkedit_args(args) -> dict:
  87    """Specifically cleans NetworkEditTool arguments.
  88    Handles both:
  89    1. Entire dict being a string
  90    2. command_arguments being a string that should be a dict
  91    """
  92    # First handle if the entire thing is a string
  93    if isinstance(args, str):
  94        start = args.find('{')
  95        end = args.rfind('}')
  96        if start == -1 or end == -1:
  97            raise ValueError(f"No dict structure found in: {args}")
  98        args = ast.literal_eval(args[start:end + 1])
  99    
 100    # Now we should have a dict. Check if command_arguments is a string
 101    if isinstance(args.get('command_arguments'), str):
 102        cmd_args = args['command_arguments']
 103        start = cmd_args.find('{')
 104        end = cmd_args.rfind('}')
 105        if start == -1 or end == -1:
 106            raise ValueError(f"No dict structure found in command_arguments: {cmd_args}")
 107        args['command_arguments'] = ast.literal_eval(cmd_args[start:end + 1])
 108    
 109    return args

 583    def _generate_diff_error(self, file_content: str, old_str: str, path: str) -> str:
 584        """
 585        Finds the best-matching region for old_str in file_content and generates a
 586        detailed error message pointing to the first character mismatch in that region.
 587        """
 588        # Use SequenceMatcher to find the best alignment of old_str within file_content
 589        s = difflib.SequenceMatcher(None, old_str, file_content, autojunk=False)
 590        match = s.find_longest_match(0, len(old_str), 0, len(file_content))
 591
 592        # Determine the start of the slice in the file to compare against.
 593        # This aligns the start of old_str with the file content based on the best match.
 594        start_in_file = max(0, match.b - match.a)
 595        end_in_file = start_in_file + len(old_str)
 596        file_slice = file_content[start_in_file:end_in_file]
 597
 598        # Now, find the first difference between old_str and the aligned file_slice
 599        mismatch_offset = -1
 600        for i, (char_old, char_slice) in enumerate(zip(old_str, file_slice)):
 601            if char_old != char_slice:
 602                mismatch_offset = i
 603                break
 604        
 605        # Handle case where the difference is due to length mismatch at the end
 606        if mismatch_offset == -1 and len(old_str) != len(file_slice):
 607            mismatch_offset = len(file_slice)
 608
 609        # If no mismatch is found at all (should be rare if count==0), provide a generic error.
 610        if mismatch_offset == -1:
 611            return (
 612                f"ERROR: Your old_str input did not appear verbatim in {path}, "
 613                "but a detailed difference could not be determined. "
 614                "Consider using this bash command to see invisible characters: python3 -c \"with open('filename', 'r') as f: print(repr(f.read()))\""
 615            )
 616
 617        # Calculate the line and column of the mismatch
 618        mismatch_abs_pos = start_in_file + mismatch_offset
 619        line_content_up_to_mismatch = file_content[:mismatch_abs_pos]
 620        line = line_content_up_to_mismatch.count('') + 1
 621        last_newline_pos = line_content_up_to_mismatch.rfind('')
 622        col = mismatch_abs_pos - last_newline_pos if last_newline_pos != -1 else mismatch_abs_pos + 1
 623
 624        # Get the mismatched characters for the report
 625        expected_char = self._represent_char(old_str[mismatch_offset]) if mismatch_offset < len(old_str) else "EOF"
 626        actual_char_in_file = self._represent_char(file_slice[mismatch_offset]) if mismatch_offset < len(file_slice) else "EOF"
 627
 628        error_message = (
 629            f"ERROR: Your old_str input did not appear verbatim in {path}."
 630            f"The closest match has a mismatch at Line {line}, Column {col}."
 631            f"Expected: '{expected_char}'"
 632            f"Received: '{actual_char_in_file}'"
 633            "This bash command to see invisible characters can be used to reveal if any hidden characters are preventing the old_str from matching: `python3 -c \"with open('filename', 'r') as f: print(repr(f.read()))\"`"
 634        )
 635
 636        return error_message

 845    def write_file(self, path: Path, content: str, target_container: str | None = None):
 846        try:
 847            # ---- local write ---------------------------------------------------
 848            if target_container is None:
 849                path.write_text(content)
 850                return
 851
 852            # ---- remote write --------------------------------------------------
 853            encoded     = base64.b64encode(content.encode()).decode()
 854            quoted_path = shlex.quote(str(path))
 855            script      = f"echo '{encoded}' | base64 --decode > {quoted_path}"
 856
 857            # Try three shells in turn: sudo-bash → bash → sh
 858            for cmd in (
 859                ["docker", "exec", "-i", target_container, "sudo", "bash", "-c", script],
 860                ["docker", "exec", "-i", target_container, "bash", "-c", script],
 861                ["docker", "exec", "-i", target_container, "sh",  "-c", script],
 862            ):
 863                result = subprocess.run(cmd, capture_output=True, text=True)
 864                if result.returncode == 0:
 865                    break  # success
 866            else:
 867                # all attempts failed
 868                raise Exception(
 869                    result.stderr.strip() or f"docker exec exited {result.returncode}"
 870                )
 871
 872        except Exception as e:
 873            raise ToolError(f"ERROR: Ran into {e} while trying to write to {path}") from None

584    def _generate_diff_error(self, file_content: str, old_str: str, path: str) -> str:
585        """
586        Finds the best-matching region for old_str in file_content and generates a
587        detailed error message pointing to the first character mismatch in that region.
588        """
589        # Use SequenceMatcher to find the best alignment of old_str within file_content
590        s = difflib.SequenceMatcher(None, old_str, file_content, autojunk=False)
591        match = s.find_longest_match(0, len(old_str), 0, len(file_content))
592
593        # Determine the start of the slice in the file to compare against.
594        # This aligns the start of old_str with the file content based on the best match.
595        start_in_file = max(0, match.b - match.a)
596        end_in_file = start_in_file + len(old_str)
597        file_slice = file_content[start_in_file:end_in_file]
598
599        # Now, find the first difference between old_str and the aligned file_slice
600        mismatch_offset = -1
601        for i, (char_old, char_slice) in enumerate(zip(old_str, file_slice)):
602            if char_old != char_slice:
603                mismatch_offset = i
604                break
605        
606        # Handle case where the difference is due to length mismatch at the end
607        if mismatch_offset == -1 and len(old_str) != len(file_slice):
608            mismatch_offset = len(file_slice)
609
610        # If no mismatch is found at all (should be rare if count==0), provide a generic error.
611        if mismatch_offset == -1:
612            return (
613                f"ERROR: Your old_str input did not appear verbatim in {path}, "
614                "but a detailed difference could not be determined. "
615                "Consider using this bash command to see invisible characters: python3 -c \"with open('filename', 'r') as f: print(repr(f.read()))\""
616            )
617
618        # Calculate the line and column of the mismatch
619        mismatch_abs_pos = start_in_file + mismatch_offset
620        line_content_up_to_mismatch = file_content[:mismatch_abs_pos]
621        line = line_content_up_to_mismatch.count('') + 1
622        last_newline_pos = line_content_up_to_mismatch.rfind('')
623        col = mismatch_abs_pos - last_newline_pos if last_newline_pos != -1 else mismatch_abs_pos + 1
624
625        # Get the mismatched characters for the report
626        expected_char = self._represent_char(old_str[mismatch_offset]) if mismatch_offset < len(old_str) else "EOF"

635        )
