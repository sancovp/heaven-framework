 195            content=summary,

 227            if hasattr(self.registry_builder, 'add_summary'):
 228                summary_id = self.registry_builder.add_summary(
 229                    summary=summary,
 230                    level=level,
 231                    source_paths=source_paths,
 232                    concepts=concepts,
 233                    parents=parents,
 234                    summary_id=registry_key,

 236                )
 237                self._registry_keys.append(summary_id)
 238                print(f"Added summary to registry with ID: {summary_id}")
 239            else:
 240                # Fallback to standard add method if add_summary isn't available
 241                try:
 242                    self.registry_builder.add(registry_key, {
 243                        "content": summary,
 244                        "level": level,
 245                        "source_paths": source_paths,

 289        """Create batches of summaries that fit within token limits."""

 300            else:
 301                current_batch.append(entry)
 302                current_size += entry_size

 304        if current_batch:
 305            batches.append(current_batch)

 390    async def process_level(self, level: int) -> Optional[str]:
 391        """
 392        Process all summaries at a level and generate a higher-level summary if needed.
 393        
 394        Args:
 395            level: Level to process
 396            
 397        Returns:
 398            New summary text if generated, None otherwise
 399        """
 400        # Check if we need to summarize this level
 401        total_chars = self._calculate_tokens_at_level(level)
 402        threshold = BASE_LEVEL_TOKEN_THRESHOLD if level == 0 else META_LEVEL_TOKEN_THRESHOLD
 403        print(f"Level {level} has {total_chars} characters (threshold: {threshold})")
 404        
 405        if not self._should_summarize_level(level):
 406            print(f"Level {level} does not need summarization yet")
 407            return None
 408        
 409        print(f"Summarizing level {level}...")
 410        
 411        # Create injection history
 412        history, history_id = await self._create_injection_history(level)
 413        print(f"Created injection history: {history_id}")
 414        
 415        # Create a MetaSummarizerAgent
 416        agent = MetaSummarizerAgent()
 417        
 418        # Get the content from all summaries at this level
 419        summaries_content = "\n\n".join([entry.get_formatted_content() for entry in self._get_summaries_by_level(level)])
 420        
 421        # Use the agent's create_meta_summary method directly
 422        summary = await agent.create_meta_summary(summaries_content)
 423            
 424        # If the agent didn't produce a valid summary, provide a minimal fallback
 425        if not summary or len(summary.strip()) < 10:
 426            print("Warning: Agent didn't produce a valid summary. Using minimal fallback.")
 427            summary = f"# Level {level+1} Summary\nSummary of {len(self._get_summaries_by_level(level))} level {level} summaries."
 428        
 429        print(f"Generated level {level+1} summary with {len(summary)} characters")
 430        
 431        # Get source paths from all entries at this level
 432        source_paths = []
 433        # Collect all concepts from lower level
 434        all_concepts = []
 435        
 436        for entry in self._get_summaries_by_level(level):
 437            source_paths.extend(entry.source_paths)
 438            
 439            # Collect concepts from metadata
 440            if hasattr(entry, 'metadata') and entry.metadata:
 441                if 'concepts' in entry.metadata and isinstance(entry.metadata['concepts'], list):
 442                    all_concepts.extend(entry.metadata['concepts'])
 443        
 444        # Deduplicate concepts and count frequencies
 445        concept_counts = {}
 446        for concept in all_concepts:
 447            if concept in concept_counts:
 448                concept_counts[concept] += 1
 449            else:
 450                concept_counts[concept] = 1
 451        
 452        # Keep only concepts that appear multiple times or are important
 453        important_concepts = ["architecture", "design", "patterns", "components"]
 454        propagated_concepts = [
 455            concept for concept, count in concept_counts.items() 
 456            if count > 1 or concept in important_concepts
 457        ]
 458        
 459        # Ensure we have at least one concept
 460        if not propagated_concepts:
 461            propagated_concepts = ["general"]
 462            
 463        print(f"Propagating concepts to level {level+1}: {propagated_concepts}")
 464        
 465        # Add this new summary to the next level with propagated concepts
 466        self.add_summary(
 467            summary=summary,
 468            level=level + 1,
 469            source_paths=list(set(source_paths)),  # Deduplicate
 470            metadata={
 471                "source_level": level,
 472                "concepts": propagated_concepts,
 473                "parent_concepts": all_concepts
 474            }
 475        )
 476        
 477        print(f"Added new summary to level {level+1}")
 478        return summary

 499                # No summary generated at this level, move to next

 575    def get_summary_by_concept(self, concept: str, level: Optional[int] = None, exact_match: bool = False) -> List[Dict[str, Any]]:
 576        """
 577        Get all summaries tagged with a specific concept.
 578        
 579        Args:
 580            concept: The concept to search for
 581            level: Optional level to filter by
 582            exact_match: If True, only return summaries explicitly tagged with this concept
 583                        If False, also look for the concept in content
 584            
 585        Returns:
 586            List of summary dictionaries with content, level, relevance, etc.
 587        """
 588        results = []
 589        
 590        # Manual search through summaries, prioritizing explicit tags
 591        for summary_level, entries in self.summaries.items():
 592            # Skip if level filter doesn't match
 593            if level is not None and summary_level != level:
 594                continue
 595                
 596            for entry in entries:
 597                relevance_score = 0
 598                concept_matches = []
 599                
 600                # Check if explicitly tagged with this concept in metadata
 601                if hasattr(entry, 'metadata') and entry.metadata:
 602                    if 'concepts' in entry.metadata and isinstance(entry.metadata['concepts'], list):
 603                        if concept.lower() in [c.lower() for c in entry.metadata['concepts']]:
 604                            # Directly tagged is highest relevance
 605                            relevance_score = 10
 606                            concept_matches.append(f"Tagged with '{concept}'")
 607                
 608                # If not exact match or no explicit tag found, check content
 609                if not exact_match or relevance_score == 0:
 610                    # Check if concept is mentioned in content
 611                    if entry.content and concept.lower() in entry.content.lower():
 612                        # Content match has medium relevance
 613                        content_relevance = 5
 614                        # If there's a direct match (including word boundaries), add more relevance
 615                        import re
 616                        if re.search(r'\b' + re.escape(concept.lower()) + r'\b', entry.content.lower()):
 617                            content_relevance += 2
 618                        
 619                        # Count occurrences of the concept

 289    def _create_batch(self, entries: List[RecursiveSummaryEntry], max_chars: int = 70000) -> List[List[RecursiveSummaryEntry]]:
 290        """Create batches of summaries that fit within token limits."""

 295        for entry in entries:

 298                batches.append(current_batch)
 299                current_batch = [entry]
 300                current_size = entry_size
 301            else:
 302                current_batch.append(entry)
 303                current_size += entry_size
 304        
 305        if current_batch:
 306            batches.append(current_batch)
 307        
 308        return batches

 391    async def process_level(self, level: int) -> Optional[str]:
 392        """
 393        Process all summaries at a level and generate a higher-level summary if needed.
 394        
 395        Args:
 396            level: Level to process
 397            
 398        Returns:
 399            New summary text if generated, None otherwise
 400        """
 401        # Check if we need to summarize this level
 402        total_chars = self._calculate_tokens_at_level(level)
 403        threshold = BASE_LEVEL_TOKEN_THRESHOLD if level == 0 else META_LEVEL_TOKEN_THRESHOLD
 404        print(f"Level {level} has {total_chars} characters (threshold: {threshold})")
 405        
 406        if not self._should_summarize_level(level):
 407            print(f"Level {level} does not need summarization yet")
 408            return None
 409        
 410        print(f"Summarizing level {level}...")
 411        
 412        # Create injection history
 413        history, history_id = await self._create_injection_history(level)
 414        print(f"Created injection history: {history_id}")
 415        
 416        # Create a MetaSummarizerAgent
 417        agent = MetaSummarizerAgent()
 418        
 419        # Get the content from all summaries at this level
 420        summaries_content = "\n\n".join([entry.get_formatted_content() for entry in self._get_summaries_by_level(level)])
 421        
 422        # Use the agent's create_meta_summary method directly
 423        summary = await agent.create_meta_summary(summaries_content)
 424            
 425        # If the agent didn't produce a valid summary, provide a minimal fallback
 426        if not summary or len(summary.strip()) < 10:
 427            print("Warning: Agent didn't produce a valid summary. Using minimal fallback.")
 428            summary = f"# Level {level+1} Summary\nSummary of {len(self._get_summaries_by_level(level))} level {level} summaries."
 429        
 430        print(f"Generated level {level+1} summary with {len(summary)} characters")
 431        
 432        # Get source paths from all entries at this level
 433        source_paths = []
 434        # Collect all concepts from lower level
 435        all_concepts = []
 436        
 437        for entry in self._get_summaries_by_level(level):
 438            source_paths.extend(entry.source_paths)
 439            
 440            # Collect concepts from metadata
 441            if hasattr(entry, 'metadata') and entry.metadata:
 442                if 'concepts' in entry.metadata and isinstance(entry.metadata['concepts'], list):
 443                    all_concepts.extend(entry.metadata['concepts'])
 444        
 445        # Deduplicate concepts and count frequencies
 446        concept_counts = {}
 447        for concept in all_concepts:
 448            if concept in concept_counts:
 449                concept_counts[concept] += 1
 450            else:
 451                concept_counts[concept] = 1
 452        
 453        # Keep only concepts that appear multiple times or are important
 454        important_concepts = ["architecture", "design", "patterns", "components"]
 455        propagated_concepts = [
 456            concept for concept, count in concept_counts.items() 
 457            if count > 1 or concept in important_concepts
 458        ]
 459        
 460        # Ensure we have at least one concept
 461        if not propagated_concepts:
 462            propagated_concepts = ["general"]
 463            
 464        print(f"Propagating concepts to level {level+1}: {propagated_concepts}")
 465        
 466        # Add this new summary to the next level with propagated concepts
 467        self.add_summary(
 468            summary=summary,
 469            level=level + 1,
 470            source_paths=list(set(source_paths)),  # Deduplicate
 471            metadata={
 472                "source_level": level,
 473                "concepts": propagated_concepts,
 474                "parent_concepts": all_concepts
 475            }
 476        )
 477        
 478        print(f"Added new summary to level {level+1}")
 479        return summary

 481    async def process_all_levels(self) -> Dict[int, List[str]]:
 482        """
 483        Process all levels recursively, starting from the lowest.
 484        
 485        Returns:
 486            Dict mapping level -> list of summary texts generated
 487        """
 488        results = {}
 489        current_level = 0
 490        
 491        # Process each level until no more summaries are generated
 492        while current_level in self.summaries:
 493            summary = await self.process_level(current_level)
 494            if summary:
 495                if current_level + 1 not in results:
 496                    results[current_level + 1] = []
 497                results[current_level + 1].append(summary)
 498                current_level += 1
 499            else:
 500                # No summary generated at this level, move to next
 501                current_level += 1
 502        
 503        return results

 576    def get_summary_by_concept(self, concept: str, level: Optional[int] = None, exact_match: bool = False) -> List[Dict[str, Any]]:
 577        """
 578        Get all summaries tagged with a specific concept.
 579        
 580        Args:
 581            concept: The concept to search for
 582            level: Optional level to filter by
 583            exact_match: If True, only return summaries explicitly tagged with this concept
 584                        If False, also look for the concept in content
 585            
 586        Returns:
 587            List of summary dictionaries with content, level, relevance, etc.
 588        """
 589        results = []
 590        
 591        # Manual search through summaries, prioritizing explicit tags
 592        for summary_level, entries in self.summaries.items():
 593            # Skip if level filter doesn't match
 594            if level is not None and summary_level != level:
 595                continue
 596                
 597            for entry in entries:
 598                relevance_score = 0
 599                concept_matches = []
 600                
 601                # Check if explicitly tagged with this concept in metadata
 602                if hasattr(entry, 'metadata') and entry.metadata:
 603                    if 'concepts' in entry.metadata and isinstance(entry.metadata['concepts'], list):
 604                        if concept.lower() in [c.lower() for c in entry.metadata['concepts']]:
 605                            # Directly tagged is highest relevance
 606                            relevance_score = 10
 607                            concept_matches.append(f"Tagged with '{concept}'")
 608                
 609                # If not exact match or no explicit tag found, check content
 610                if not exact_match or relevance_score == 0:
 611                    # Check if concept is mentioned in content
 612                    if entry.content and concept.lower() in entry.content.lower():
 613                        # Content match has medium relevance
 614                        content_relevance = 5
 615                        # If there's a direct match (including word boundaries), add more relevance
 616                        import re
 617                        if re.search(r'\b' + re.escape(concept.lower()) + r'\b', entry.content.lower()):
 618                            content_relevance += 2
 619                        
 620                        # Count occurrences of the concept
 621                        occurrences = entry.content.lower().count(concept.lower())
 622                        if occurrences > 1:
 623                            content_relevance += min(occurrences, 3)  # Cap at +3 for multiple occurrences
 624                            
 625                        if relevance_score < content_relevance:
 626                            relevance_score = content_relevance
 627                            concept_matches.append(f"Content mentions '{concept}' {occurrences} times")
 628                
 629                # Skip if no relevance and exact matching
 630                if relevance_score == 0 and exact_match:
 631                    continue
 632                
 633                # Skip if no relevance at all
 634                if relevance_score == 0:
 635                    continue
 636                    
 637                # Prepare the result
 638                entry_id = f"summary_{summary_level}_{entry.timestamp.strftime('%Y%m%d_%H%M%S')}"
 639                result = {
 640                    "level": summary_level,
 641                    "id": entry_id,
 642                    "content": entry.content,
 643                    "source_paths": entry.source_paths,
 644                    "timestamp": entry.timestamp.isoformat(),
 645                    "relevance_score": relevance_score,
 646                    "match_details": ", ".join(concept_matches)
 647                }
 648                
 649                # Add metadata
 650                if hasattr(entry, 'metadata') and entry.metadata:
 651                    result["metadata"] = entry.metadata
 652                
 653                results.append(result)
 654        
 655        # Try registry method if available
 656        if (hasattr(self, 'registry_builder') and 
 657            self.registry_builder and 
 658            hasattr(self.registry_builder, 'get_summaries_by_concept')):
 659            try:
 660                registry_summaries = self.registry_builder.get_summaries_by_concept(concept)
 661                
 662                # Filter by level if specified
 663                if level is not None:
 664                    registry_summaries = [s for s in registry_summaries if s.get('level') == level]
 665                    
 666                # Format and add the summaries
 667                for summary in registry_summaries:
 668                    # Registry summaries are considered highly relevant
 669                    relevance_score = 9
 670                    result = {
 671                        "level": summary.get('level'),
 672                        "id": summary.get('id'),
 673                        "content": summary.get('content', ''),
 674                        "source_paths": summary.get('source_paths', []),
 675                        "relevance_score": relevance_score,
 676                        "match_details": "Registry match",
 677                        "from_registry": True
 678                    }
 679                    results.append(result)
 680            except (AttributeError, TypeError):
 681                pass
 682        
 683        # Sort by relevance score (highest first)
 684        results.sort(key=lambda x: (-x["relevance_score"], x["level"]))
 685        
 686        # Format the results for return
 687        formatted_results = []
 688        for result in results:
 689            # Create a header with level, id, and relevance
 690            header = f"## Level {result['level']} Summary | {result['id']} | Relevance: {result['relevance_score']}/10\n"
 691            header += f"Sources: {', '.join(result['source_paths'])}\n"
 692            if "match_details" in result:
 693                header += f"Match: {result['match_details']}\n"
 694            
 695            # Format all the result info into a dictionary
 696            formatted_results.append({
 697                "formatted_text": f"{header}\n{result['content']}",
 698                "raw": result
 699            })
 700            
 701        return formatted_results

 703    def get_related_summaries(self, summary_id: str, max_results: int = 5) -> List[Dict[str, Any]]:

 711        Returns:

 713        """

 727                    # Get concepts from the target entry

 731                    break
 732            if target_entry:
 733                break
 734
