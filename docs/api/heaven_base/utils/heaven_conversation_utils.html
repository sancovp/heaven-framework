heaven_base.utils.heaven_conversation_utils

HEAVEN Conversation Management Utilities

These utilities manage conversations as collections of related history snapshots. Each conversation tracks the terminal history_ids from continued chains.

Directory structure: /conversations/{month}/{day}/{datetime}.json

Each conversation file contains: { "conversation_id": "2025_08_04_17_30_00", "title": "User-provided title or auto-generated", "created_datetime": "2025-08-04T17:30:00.123456", "last_updated": "2025-08-04T17:45:15.789012", "history_chain": [ "2025_08_04_17_30_00_agent_continued_3", // Last in continuation chain "2025_08_04_17_45_15_agent" // Single history (no continuations) ], "metadata": { "agent_name": "ChatAgent", "total_exchanges": 5, "tags": ["astronomy", "learning"] } }

NOTE: These utilities are designed to be moved to heaven-framework eventually.

 56    @staticmethod
 57    def _get_conversation_file_path(conversation_id: str) -> str:
 58        """Get the file path for a conversation ID."""
 59        conversations_dir = ConversationManager._get_conversations_dir()
 60        
 61        # Extract month and day from conversation_id (format: YYYY_MM_DD_HH_MM_SS)
 62        parts = conversation_id.split("_")
 63        if len(parts) >= 3:
 64            month = parts[1]  # MM
 65            day = parts[2]    # DD
 66        else:
 67            # Fallback to current month/day
 68            now = datetime.now()
 69            month = now.strftime("%m")
 70            day = now.strftime("%d")
 71        
 72        # Create directory structure
 73        conv_path = os.path.join(conversations_dir, month, day)
 74        os.makedirs(conv_path, exist_ok=True)
 75        
 76        return os.path.join(conv_path, f"{conversation_id}.json")

 78    @staticmethod
 79    def start_conversation(title: str, first_history_id: str, agent_name: str, 
 80                          tags: Optional[List[str]] = None) -> Dict[str, Any]:
 81        """
 82        Start a new conversation with the first history.
 83        
 84        Args:
 85            title: Human-readable conversation title
 86            first_history_id: ID of the first history in this conversation
 87            agent_name: Name of the agent involved
 88            tags: Optional tags for categorization
 89            
 90        Returns:
 91            Dictionary with conversation data
 92        """
 93        conversation_id = ConversationManager._generate_conversation_id()
 94        now = datetime.now()
 95        
 96        conversation_data = {
 97            "conversation_id": conversation_id,
 98            "title": title,
 99            "created_datetime": now.isoformat(),
100            "last_updated": now.isoformat(),
101            "history_chain": [first_history_id],
102            "metadata": {
103                "agent_name": agent_name,
104                "total_exchanges": 1,
105                "tags": tags or []
106            }
107        }
108        
109        # Save to file
110        file_path = ConversationManager._get_conversation_file_path(conversation_id)
111        with open(file_path, 'w') as f:
112            json.dump(conversation_data, f, indent=2)
113        
114        return conversation_data

116    @staticmethod
117    def continue_conversation(conversation_id: str, new_history_id: str) -> Dict[str, Any]:
118        """
119        Continue an existing conversation by adding a new history.
120        
121        Args:
122            conversation_id: ID of the conversation to continue
123            new_history_id: ID of the new history to add
124            
125        Returns:
126            Updated conversation data
127            
128        Raises:
129            FileNotFoundError: If conversation doesn't exist
130        """
131        file_path = ConversationManager._get_conversation_file_path(conversation_id)
132        
133        if not os.path.exists(file_path):
134            raise FileNotFoundError(f"Conversation {conversation_id} not found")
135        
136        # Load existing conversation
137        with open(file_path, 'r') as f:
138            conversation_data = json.load(f)
139        
140        # Add new history to chain
141        conversation_data["history_chain"].append(new_history_id)
142        conversation_data["last_updated"] = datetime.now().isoformat()
143        conversation_data["metadata"]["total_exchanges"] += 1
144        
145        # Save updated conversation
146        with open(file_path, 'w') as f:
147            json.dump(conversation_data, f, indent=2)
148        
149        return conversation_data

175    @staticmethod
176    def list_conversations(limit: Optional[int] = None) -> List[Dict[str, Any]]:
177        """
178        List all conversations, sorted by last_updated (newest first).
179        
180        Args:
181            limit: Optional limit on number of conversations to return
182            
183        Returns:
184            List of conversation data dictionaries
185        """
186        conversations_dir = ConversationManager._get_conversations_dir()
187        
188        # Find all conversation files
189        pattern = os.path.join(conversations_dir, "**", "*.json")
190        conv_files = glob.glob(pattern, recursive=True)
191        
192        conversations = []
193        for file_path in conv_files:
194            try:
195                with open(file_path, 'r') as f:
196                    conv_data = json.load(f)
197                    conversations.append(conv_data)
198            except Exception as e:
199                print(f"Error loading conversation file {file_path}: {e}")
200                continue
201        
202        # Sort by last_updated (newest first)
203        conversations.sort(key=lambda x: x.get("last_updated", ""), reverse=True)
204        
205        if limit:
206            conversations = conversations[:limit]
207        
208        return conversations

250    @staticmethod
251    def get_conversation_latest_history(conversation_id: str) -> Optional[str]:
252        """
253        Get the latest/terminal history ID from a conversation for context loading.
254        This avoids loading redundant intermediate snapshots.
255        
256        Args:
257            conversation_id: ID of the conversation
258            
259        Returns:
260            Latest history ID (most complete snapshot), or None if not found
261        """

264            return None

269    @staticmethod
270    def search_conversations(query: str, search_fields: Optional[List[str]] = None) -> List[Dict[str, Any]]:
271        """
272        Search conversations by text in specified fields.
273        
274        Args:
275            query: Search query string
276            search_fields: Fields to search in (default: ["title", "tags"])
277            
278        Returns:
279            List of matching conversation data dictionaries
280        """
281        if search_fields is None:
282            search_fields = ["title", "tags"]
283        
284        query_lower = query.lower()
285        all_conversations = ConversationManager.list_conversations()
286        
287        matching_conversations = []
288        for conv in all_conversations:
289            match_found = False
290            
291            for field in search_fields:
292                if field == "title" and query_lower in conv.get("title", "").lower():
293                    match_found = True
294                    break
295                elif field == "tags":
296                    tags = conv.get("metadata", {}).get("tags", [])
297                    if any(query_lower in tag.lower() for tag in tags):
298                        match_found = True
299                        break
300                elif field in conv and query_lower in str(conv[field]).lower():
301                    match_found = True
302                    break
303            
304            if match_found:
305                matching_conversations.append(conv)
306        
307        return matching_conversations

271    def search_conversations(query: str, search_fields: Optional[List[str]] = None) -> List[Dict[str, Any]]:
272        """
273        Search conversations by text in specified fields.
274        
275        Args:
276            query: Search query string
277            search_fields: Fields to search in (default: ["title", "tags"])
278            
279        Returns:
280            List of matching conversation data dictionaries
281        """
282        if search_fields is None:
283            search_fields = ["title", "tags"]
284        
285        query_lower = query.lower()
286        all_conversations = ConversationManager.list_conversations()
287        
288        matching_conversations = []
289        for conv in all_conversations:
290            match_found = False
291            
292            for field in search_fields:
293                if field == "title" and query_lower in conv.get("title", "").lower():
294                    match_found = True
295                    break
296                elif field == "tags":
297                    tags = conv.get("metadata", {}).get("tags", [])
298                    if any(query_lower in tag.lower() for tag in tags):
299                        match_found = True
300                        break
301                elif field in conv and query_lower in str(conv[field]).lower():
302                    match_found = True
303                    break
304            
305            if match_found:
306                matching_conversations.append(conv)
307        
308        return matching_conversations
