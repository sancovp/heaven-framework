  75    """Summarize a single iteration with simplified user/assistant/pain_points format."""
  76    
  77    # Format concept tags for display
  78    tags_display = ""
  79    if concept_tags:
  80        tags_display = "\n**Concept Tags:**\n"
  81        for tag in concept_tags:
  82            if isinstance(tag, dict):
  83                keyword = tag.get('keyword', 'Unknown')
  84                description = tag.get('description', 'No description')
  85                tags_display += f"- {keyword}: {description}\n"
  86    
  87    # Format pain points for display
  88    pain_display = ""
  89    if pain_points:
  90        pain_display = "\n**Pain Points:**\n"
  91        for pain in pain_points:
  92            pain_display += f"- {pain}\n"
  93    
  94    # Format files touched for display
  95    files_display = ""
  96    if files_touched:
  97        files_display = "\n**Files Touched:**\n"
  98        for file_path in files_touched:
  99            files_display += f"- {file_path}\n"

 102
 103**User:** {user}

 496def auto_summarize_flag(history_id_or_obj: Union[str, History], summarize_at: int = 450000) -> bool:
 497    """
 498    Check if history content exceeds the token threshold for summarization.
 499    
 500    Args:
 501        history_id_or_obj: Either a history_id string or History object
 502        summarize_at: Character threshold (roughly 150k tokens at default of 450000 chars)
 503                      Approximating 3 characters per token
 504    
 505    Returns:
 506        bool: True if summarization is needed, False otherwise
 507    """
 508    # Handle both history_id and History object
 509    if isinstance(history_id_or_obj, str):
 510        history = History.load_from_id(history_id_or_obj)
 511    else:
 512        history = history_id_or_obj
 513    
 514    # Calculate total character count (rough approximation of tokens)
 515    total_chars = 0
 516    for msg in history.messages:
 517        if isinstance(msg, BaseMessage):
 518            total_chars += len(str(msg.content))
 519    
 520    return total_chars > summarize_at
